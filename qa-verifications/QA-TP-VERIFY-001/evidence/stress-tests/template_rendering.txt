=== ST-3: Template rendering ===
render_template() {
  local template_file="$1"
  shift
  # Remaining args are key=value pairs
  local content
  content=$(cat "$template_file")
  while [ $# -gt 0 ]; do
    local key="${1%%=*}"
    local value="${1#*=}"
    content="${content//\{\{${key}\}\}/$value}"
    shift
  done
  echo "$content"
}
