@app.post("/api/deals/{deal_id}/archive")
async def archive_deal(
    deal_id: str,
    pool: asyncpg.Pool = Depends(get_db)
):
    """Archive a deal by moving it to archived stage."""
    async with pool.acquire() as conn:
        # Verify deal exists and is not already archived
        deal = await conn.fetchrow(
            "SELECT deal_id, stage FROM zakops.deals WHERE deal_id = $1 AND deleted = FALSE",
            deal_id
        )
        if not deal:
            raise HTTPException(status_code=404, detail=f"Deal {deal_id} not found")

        if deal['stage'] == 'archived':
            return {"success": True, "message": "Deal already archived", "deal_id": deal_id}

        prev_stage = deal['stage']

        # Update to archived stage
        await conn.execute(
            """
            UPDATE zakops.deals
            SET stage = 'archived', updated_at = NOW()
            WHERE deal_id = $1
            """,
            deal_id
        )

        # Record event
        await conn.execute(
            """
            INSERT INTO zakops.deal_events (deal_id, event_type, event_source, actor, payload)
            VALUES ($1, 'deal_archived', 'api', 'ui_user', $2::jsonb)
            """,
            deal_id,
            json.dumps({"previous_stage": prev_stage})
        )

    return {"success": True, "deal_id": deal_id, "previous_stage": prev_stage}
