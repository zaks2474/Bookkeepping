CREATE INDEX IF NOT EXISTS idx_threads_user ON chat_threads(user_id);
CREATE INDEX IF NOT EXISTS idx_threads_deal ON chat_threads(deal_id) WHERE deal_id IS NOT NULL;
CREATE INDEX IF NOT EXISTS idx_threads_active ON chat_threads(user_id, last_active DESC) WHERE deleted = FALSE;
CREATE INDEX IF NOT EXISTS idx_threads_deleted ON chat_threads(deleted_at) WHERE deleted = TRUE;
