import { z } from 'zod';

/**
 * Safely load data from localStorage with Zod validation.
 *
 * RT-STORE-1 Requirements:
 * 1. If parse fails: DELETE the key, return fallback
 * 2. If schema fails: DELETE the key, return fallback
 * 3. If version mismatch: DELETE the key, return fallback
 * 4. Console.warn on quarantine (not error)
 *
 * @param key - localStorage key
 * @param schema - Zod schema to validate against
 * @param options - { version: expected version number, fallback: default value }
 */
export function loadFromStorage<T>(
  key: string,
  schema: z.ZodSchema<T>,
  options: { version: number; fallback: T }
): T {
  try {
    const raw = localStorage.getItem(key);
    if (!raw) return options.fallback;

    // Try to parse JSON
    let parsed: unknown;
    try {
      parsed = JSON.parse(raw);
    } catch {
      console.warn(`[storage-utils] JSON parse failed for ${key}, quarantining`);
